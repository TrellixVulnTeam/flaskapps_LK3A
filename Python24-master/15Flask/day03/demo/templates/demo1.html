{# macro宏，可以看作是是jinja2的一个函数，返回一个模板或者html代码 #} {# flask模板代码的复用使用macro宏, 使用方法就跟function函数一样 #} {# 定义宏，相当于定义一个函数，在使用的时候直接调用该宏，传入不同的参数就可以了 #} {% macro my_input(label='', type='text', name='', value='') %}
{% endmacro %}
{{ my_input('用户名：', name='username') }} {{ my_input('密码：', type='password', name='password') }} {{ my_input('确认密码：', type='password', name='password2') }} {{ my_input(type='submit', value='注册') }}
{# 导入macro, 把所有的macro抽取到一个macro.html #} {% import 'macro.html' as my_macro %}
{{ my_macro.my_input('用户名：', name='username') }} {{ my_macro.my_input('密码：', type='password', name='password') }} {{ my_macro.my_input('确认密码：', type='password', name='password2') }} {{ my_macro.my_input(type='submit', value='注册') }}
